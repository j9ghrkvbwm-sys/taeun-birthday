  <script>

    const TOTAL = 30;

    let count = 0;

    let locked = false;


    const cakeZone = document.getElementById("cakeZone");

    const cake = document.getElementById("cake");

    const spoon = document.getElementById("spoon");

    const counter = document.getElementById("counter");

    const message = document.getElementById("message");

    const restart = document.getElementById("restart");

    const bites = [...document.querySelectorAll(".bite")];


    function playSound() {

      try {

        const AudioContext = window.AudioContext || window.webkitAudioContext;

        const ctx = new AudioContext();

        const osc = ctx.createOscillator();

        const gain = ctx.createGain();


        osc.type = "sine";

        osc.frequency.setValueAtTime(520, ctx.currentTime);

        osc.frequency.exponentialRampToValueAtTime(760, ctx.currentTime + 0.08);


        gain.gain.setValueAtTime(0.05, ctx.currentTime);

        gain.gain.exponentialRampToValueAtTime(0.001, ctx.currentTime + 0.12);


        osc.connect(gain);

        gain.connect(ctx.destination);

        osc.start();

        osc.stop(ctx.currentTime + 0.13);

      } catch (error) {

        // 브라우저가 오디오를 막아도 게임은 정상 진행

      }

    }


    function animateSpoon() {

      spoon.classList.remove("animate");

      void spoon.offsetWidth;

      spoon.classList.add("animate");

    }


    function updateCake() {

      const visibleBites = Math.min(8, Math.floor(count / 3.5));


      bites.forEach((bite, index) => {

        bite.classList.toggle("show", index < visibleBites);

      });


      const scale = Math.max(0.58, 1 - count * 0.012);

      cake.style.transform = `scale(${scale})`;


      if (count >= 24) cake.style.opacity = "0.72";

      if (count >= 28) cake.style.opacity = "0.38";

    }


    function eatCake() {

      if (locked || count >= TOTAL) return;


      count += 1;

      counter.textContent = `먹은 횟수: ${count} / ${TOTAL}`;


      playSound();

      animateSpoon();

      updateCake();


      if (count === TOTAL) {

        locked = true;

        setTimeout(finishGame, 450);

      }

    }


    function finishGame() {

      cake.classList.add("hidden");

      launchConfetti();

      message.classList.add("show");

    }


    function launchConfetti() {

      const icons = ["■", "●", "★", "✦", "◆"];


      for (let i = 0; i < 55; i += 1) {

        const piece = document.createElement("span");

        piece.className = "confetti";

        piece.textContent = icons[Math.floor(Math.random() * icons.length)];

        piece.style.left = `${Math.random() * 100}%`;

        piece.style.top = `${-20 - Math.random() * 35}px`;

        piece.style.fontSize = `${10 + Math.random() * 14}px`;

        piece.style.animationDelay = `${Math.random() * 0.65}s`;
